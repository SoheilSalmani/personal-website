     const citiesCol = collection(db, "cities")

3. Initialize the Admin SDK (Node.js, Java and Python).

   <Solution>

   ```sh
   npm install firebase-admin
   ```

   ```js nu
   import { initializeApp } from "firebase-admin/app"
   import { getProjectManagement } from "firebase-admin/project-management"

   const app = initializeApp()
   const projectManagement = getProjectManagement(app)
   projectManagement.setDisplayName("my-firebase-project")
   ```

   ```xml fp=pom.xml
   <dependency>
     <groupId>org.slf4j</groupId>
     <artifactId>slf4j-simple</artifactId>
     <version>1.7.35</version>
   </dependency>
   <dependency>
     <groupId>com.google.firebase</groupId>
     <artifactId>firebase-admin</artifactId>
     <version>8.1.0</version>
   </dependency>
   <dependency>
     <groupId>com.google.auth</groupId>
     <artifactId>google-auth-library-oauth2-http</artifactId>
     <version>1.4.0</version>
   </dependency>
   ```

   ```java nu
   package com.mycompany.app;

   import com.google.firebase.FirebaseApp;
   import com.google.firebase.projectmanagement.FirebaseProjectManagement;
   import com.google.firebase.projectmanagement.FirebaseProjectManagementException;

   public class App {
       public static void main(String[] args) throws FirebaseProjectManagementException {
           FirebaseApp app = FirebaseApp.initializeApp();
           FirebaseProjectManagement projectManagement = FirebaseProjectManagement.getInstance(app);
           projectManagement.createAndroidApp("com.mycompany.app");
       }
   }
   ```

   ```sh
   pip install firebase-admin
   ```

   ```python nu
   import firebase_admin
   from firebase_admin import project_management

   app = firebase_admin.initialize_app()
   android_apps = project_management.list_android_apps()
   print([app.get_metadata().package_name for app in android_apps])
   ```

   </Solution>
